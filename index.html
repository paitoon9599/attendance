// ========================================== // 🛠️ ส่วนตั้งค่าระบบ (ปรับแต่งตรงนี้ได้เลย) // ========================================== const TARGET_LAT = 13.7563; // 📍 พิกัดละทีจูดของบริษัท const TARGET_LNG = 100.5018; // 📍 พิกัดลองจิจูดของบริษัท const MAX_DISTANCE_M = 1000000; // 📏 ระยะห่างที่อนุญาต (เมตร) const DRIVE_FOLDER_ID = "112IU1JrxeNPtLONsELebjuBbDyFjt0Gi"; // 📁 ID โฟลเดอร์เก็บรูปพนักงาน function doPost(e) { let response; try { const payload = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (payload.action === "submit_attendance") { const distance = calculateDistance(payload.lat, payload.lng, TARGET_LAT, TARGET_LNG); if (distance > MAX_DISTANCE_M) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "❌ บันทึกไม่สำเร็จ: คุณอยู่นอกพื้นที่บริษัท (ห่าง " + Math.round(distance) + " เมตร)" })).setMimeType(ContentService.MimeType.JSON); } // ⏱️ คำนวณช่วงเวลาปัจจุบัน const now = new Date(); const timeStr = Utilities.formatDate(now, "GMT+7", "HH:mm"); const todayStr = Utilities.formatDate(now, "GMT+7", "yyyy-MM-dd"); const currentHour = now.getHours(); const currentMinute = now.getMinutes(); const currentTimeValue = (currentHour * 100) + currentMinute; // 🚨 [เพิ่มระบบ: บล็อกวันเสาร์ และ วันอาทิตย์ อัตโนมัติ] 🚨 const currentDay = now.getDay(); // 0 = วันอาทิตย์, 6 = วันเสาร์ if (currentDay === 0 || currentDay === 6) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "❌ ไม่สำเร็จ: วันนี้เป็นวันเสาร์-อาทิตย์ (วันหยุดประจำสัปดาห์) ไม่ต้องลงเวลาทำงานครับ" })).setMimeType(ContentService.MimeType.JSON); } // 🚨 [ระบบตรวจสอบวันหยุดราชการ/บริษัท จากแท็บ Holidays] const holidaySheet = ss.getSheetByName("Holidays"); if (holidaySheet) { const holidayLastRow = holidaySheet.getLastRow(); if (holidayLastRow > 1) { const holidayData = holidaySheet.getRange(2, 1, holidayLastRow - 1, 1).getValues(); for (let h = 0; h < holidayData.length; h++) { if (!holidayData[h][0]) continue; const holidayDateObj = new Date(holidayData[h][0]); const holidayDateStr = Utilities.formatDate(holidayDateObj, "GMT+7", "yyyy-MM-dd"); if (todayStr === holidayDateStr) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "❌ วันนี้เป็นวันหยุดราชการ/วันหยุดบริษัท ไม่ต้องลงเวลาทำงานครับ" })).setMimeType(ContentService.MimeType.JSON); } } } } let attendanceType = ""; let attendanceNote = "-"; // 1. เงื่อนไขช่วงเวลามาทำงาน (06:00 น. ถึง 11:00 น.) if (currentTimeValue >= 600 && currentTimeValue <= 1100) { attendanceType = "ลงเวลามา"; if (currentTimeValue > 800) { attendanceNote = "ขอเข้าสาย"; } else { attendanceNote = "ปกติ"; } } // 2. เงื่อนไขช่วงเวลากลับบ้าน (13:30 น. ถึง 18:00 น.) else if (currentTimeValue >= 1330 && currentTimeValue <= 1800) { attendanceType = "ลงเวลากลับ"; if (currentTimeValue < 1530) { attendanceNote = "ขอออกก่อน"; } else { attendanceNote = "ปกติ"; } } else { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "❌ บันทึกไม่สำเร็จ: ขณะนี้เวลา " + timeStr + " น. นอกเหนือช่วงเวลาลงบันทึกระบบ (มา: 06:00-11:00 / กลับ: 13:30-18:00)" })).setMimeType(ContentService.MimeType.JSON); } // 🚨 [ระบบดักการลงเวลาซ้ำประจำวัน] let logSheet = ss.getSheetByName("Attendance_Log"); if (!logSheet) { logSheet = ss.insertSheet("Attendance_Log"); logSheet.appendRow(["วัน-เวลา", "ชื่อพนักงาน", "ประเภท", "หมายเหตุ", "ระยะห่าง (เมตร)", "พิกัด", "ลิงก์รูปภาพ"]); } else { logSheet.getRange(1, 1, 1, 7).setValues([["วัน-เวลา", "ชื่อพนักงาน", "ประเภท", "หมายเหตุ", "ระยะห่าง (เมตร)", "พิกัด", "ลิงก์รูปภาพ"]]); } const lastRow = logSheet.getLastRow(); if (lastRow > 1) { const loggedData = logSheet.getRange(2, 1, lastRow - 1, 3).getValues(); for (let i = 0; i < loggedData.length; i++) { if (!loggedData[i][0]) continue; const rowDateObj = new Date(loggedData[i][0]); const rowDateStr = Utilities.formatDate(rowDateObj, "GMT+7", "yyyy-MM-dd"); const rowEmpName = loggedData[i][1].toString().trim(); const rowType = loggedData[i][2].toString().trim(); if (rowDateStr === todayStr && rowEmpName === payload.name.trim() && rowType === attendanceType) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: "⚠️ ไม่สำเร็จ: คุณได้เคย '" + attendanceType + "' ของวันนี้ไปแล้วในระบบ!" })).setMimeType(ContentService.MimeType.JSON); } } } // 📸 บันทึกรูปภาพลง Google Drive let fileUrl = ""; if (payload.photo) { const folder = DriveApp.getFolderById(DRIVE_FOLDER_ID); let base64Data = payload.photo; if (base64Data.indexOf(",") > -1) { base64Data = base64Data.split(",")[1]; } const bytes = Utilities.base64Decode(base64Data); const timeStampStr = Utilities.formatDate(now, "GMT+7", "yyyyMMdd_HHmmss"); const fileName = payload.name + "_" + attendanceType + "_" + timeStampStr + ".jpg"; const blob = Utilities.newBlob(bytes, "image/jpeg", fileName); const file = folder.createFile(blob); fileUrl = file.getUrl(); } // บันทึกข้อมูลลงตาราง const timeStamp = Utilities.formatDate(now, "GMT+7", "yyyy-MM-dd HH:mm:ss"); logSheet.appendRow([ timeStamp, payload.name, attendanceType, attendanceNote, Math.round(distance) + " เมตร", payload.lat + "," + payload.lng, fileUrl ]); let alertMsg = "✅ " + attendanceType + "สำเร็จแล้ว!"; if (attendanceNote !== "ปกติ") { alertMsg += " (⚠️ หมายเหตุ: " + attendanceNote + ")"; } response = { status: "success", message: alertMsg }; } } catch (err) { response = { status: "error", message: "ระบบหลังบ้านขัดข้อง: " + err.toString() }; } return ContentService.createTextOutput(JSON.stringify(response)) .setMimeType(ContentService.MimeType.JSON); } function calculateDistance(lat1, lon1, lat2, lon2) { const R = 6371000; const dLat = (lat2 - lat1) * Math.PI / 180; const dLon = (lon2 - lon1) * Math.PI / 180; const a = Math.sin(dLat/2) * Math.sin(dLat/2) + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2); const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); return R * c; }
